## 🎓 EJEMPLOS DE CÓDIGO

### Usar Advanced Engine
```javascript
import AdvancedAnalysisEngine from './lib/advancedAnalysisEngine.js';

const engine = new AdvancedAnalysisEngine(rawSystemData);

// Análisis arquitectónico
const arch = engine.analyzeArchitecture();
console.log(arch.overallComplexity.degree);  // 7/10

// Deuda técnica
const debt = engine.calculateTechnicalDebt();
console.log(debt.score);                      // 68/100

// Anti-patterns
const patterns = engine.detectAntiPatterns();
console.log(patterns.length);                 // 7 patrones

// SOLID analysis
const solid = engine.analyzeSolidPrinciples();
console.log(solid.D.score);                   // 3/10 - DI pobre

// Reporte completo
const full = engine.generateFullAnalysis();
console.log(full.summary.timeline);           // "3-4 months"
```

### Usar Data Flow Analyzer
```javascript
import DataFlowAnalyzer from './lib/dataFlowAnalyzer.js';

const analyzer = new DataFlowAnalyzer(rawSystemData);

// Detectar componentes
const components = analyzer.detectComponents();
console.log(components.databases);            // Oracle, MySQL, Postgres...

// Definir flujos
const flows = analyzer.defineRealDataFlows();
console.log(flows[0].totalLatency);           // "46-219ms"

// Identificar críticos
const critical = analyzer.identifyCriticalPaths();
console.log(critical[0].failureImpact);       // "System-wide outage"

// Optimizaciones
const opts = analyzer.recommendOptimizations();
console.log(opts[0].type);                    // "CACHE_LAYER"
```
